Query-First Data Model — AstraDB Cassandra
CREATE TABLE learning_events ( student_id TEXT, -- PARTITION KEY → cùng node vật lý event_time TIMESTAMP, -- CLUSTERING KEY DESC event_id UUID, event_type TEXT, -- 'video_watch' | 'quiz_submit' | 'click' course_id TEXT, duration_sec INT, score FLOAT, progress_pct FLOAT, metadata MAP<TEXT,TEXT>, PRIMARY KEY (student_id, event_time, event_id) ) WITH CLUSTERING ORDER BY (event_time DESC) AND default_time_to_live = 7776000; -- 90 ngày
